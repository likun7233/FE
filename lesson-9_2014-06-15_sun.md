#### 语句
##### 1. if语句
    var a = 100;
    if(a > 90){
      console.log('优秀');
    }else if(a > 60){
      console.log('良好');
    }else{
      console.log('不及格');
    }
    
##### 2. do-while
    var i = 0;
    do{
        i += 2;
    }while(i < 10);
    console.log(i);
    
##### 3. while
    var i = 0;
    while (i < 10) {
        i += 2;
    }
    console.log(i);
    
##### 4. for
    var len = 10;
    for (var i = 0; i < len; i++){
        console.log(i);
    }
    
##### 5. for-in
    var man = {
        'name': 'kevin',
        'sex': 'man'
    }
    for(var key in man){
        console.log(man[key]);
    }
    
##### 6. break、continue
    var i = 0;
    for(i = 0; i < 10; i++){
        if(i == 5){
            break;
        }
        console.log(i);
    }
    
    var j = 0;
    for(j = 0; j < 10; j++){
        if(j == 5){
            continue;
        }
        console.log(j);
    }
